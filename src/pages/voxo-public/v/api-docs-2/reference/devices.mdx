## List Devices {{ tag: 'GET', label: 'https://api.voxo.co/v2/devices?tenantId={tenantId}' }}

<Row>
  <Col>

    <strong>List all devices for the provided account id</strong>

    ### Parameters

    <Properties>
      <Property name="tenantId*" type="integer">1234</Property>
    </Properties>


    ### Header

    <Properties>
      <Property name="Authorization*" type="string" >
        Bearer `{accessToken}`
      </Property>
      <Property name="Content-Type*" type="string" >application/json</Property>
    </Properties>

  </Col>
  <Col sticky>

    <CodeGroup title="Request" tag="GET" label="/devices?tenantId=1234">

    ```bash {{ title: 'cURL' }}
    curl --location -g --request GET 'https://api.voxo.co/v2/devices?tenantId=1234' \
    --header 'Content-Type: application/json' \
    --header 'Authorization: Bearer {accessToken}'
    ```
    </CodeGroup>

## Search Devices {{ tag: 'GET', label: 'https://api.voxo.co/v2/devices/search?mac={deviceMac}&?name={deviceName}' }}

<Row>
  <Col>

    <strong>Search devices across all accounts by mac and/or name</strong>

    ### Parameters

    <p>
      Either name of mac search is required. MAC can be partial, but must follow the general format shown
    </p>

    <Properties>
      <Property name="name" type="integer">ayed</Property>
      <Property name="mac" type="string (MAC)">XX:XX:XX:XX:XX:XX</Property>
    </Properties>


    ### Header

    <Properties>
      <Property name="Authorization*" type="string" >
        Bearer `{accessToken}`
      </Property>
      <Property name="Content-Type*" type="string" >application/json</Property>
    </Properties>

  </Col>
